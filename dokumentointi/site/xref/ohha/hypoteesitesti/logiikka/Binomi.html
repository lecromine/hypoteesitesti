3   import org.apache.commons.math3.stat.inference.AlternativeHypothesis;
4   import org.apache.commons.math3.stat.inference.BinomialTest;

6   public class Binomi {
7   
8       double parvo;
9       BinomialTest binomi = new BinomialTest();
10  
11      /**
12       * teeTesti suorittaa testin.
13       *
14       * Jos parametri p = 0.5, on kyseessä sign -testi (kaksisuuntainen tapaus).
15       * Muutoin tehdään binomitesti.
16       *
17       * @param n diskreetin aineiston otoskoko
18       * @param k diskreetin aineiston onnistumisten lukumäärä
19       * @param p diskreetin aineiston nollahypoteesin parametri
20       * @param suunta diskreetin aineiston suunta: kaksisuuntainen, suurempaa
21       * kuin, pienempää kuin
22       * @return parvo palauttaa testin mukaisen p-arvon sitä kutsuneelle luokalle
23       */
24      public double teeBinomiTesti(int n, int k, double p, int suunta) {
25  
26          if (n <= 0) {
27              return -1;
28          } else if (k < 0) {
29              return -1;
30          } else if (k > n) {
31              return -1;
32          } else if (p <= 0) {
33              return -1;
34          } else if (p >= 1) {
35              return -1;
36          } else if (suunta < 1) {
37              return -1;
38          } else if (suunta > 3) {
39              return -1;
40          } else {
41  
42              parvo = 0;
43  
44              switch (suunta) {
45                  case 1:
46                      parvo = binomi.binomialTest(n, k, p, AlternativeHypothesis.GREATER_THAN);
47                      break;
48                  case 2:
49                      parvo = binomi.binomialTest(n, k, p, AlternativeHypothesis.LESS_THAN);
50                      break;
51                  case 3:
52                      if (p == 0.5) {
53                          parvo = binomi.binomialTest(n, k, p, AlternativeHypothesis.TWO_SIDED);
54                      } else {
55                          parvo = binomiTestiYleisesti(n, k, p);
56                      }
57                      break;
58                  default:
59                      break;
60              }
61  
62              return parvo;
63          }
64      }
65  
66      /**
67       * Tämä metodi ajetaan silloin, kun parametri p != 0.5, sillä
68       * math3-kirjaston binomitesti TWO_SIDED -tapauksessa tuottaa vääriä arvoja.
69       *
70       * @param n     otoskoko
71       * @param k     onnistumisten lkm
72       * @param p     parametri
73       * @return  parvo
74       */
75      public double binomiTestiYleisesti(int n, int k, double p) {
76  
77          double odotusarvo = n * p;
78  
79          int alaraja = (int) odotusarvo - Math.abs(k - (int) odotusarvo);
80          int ylaraja = (int) odotusarvo + Math.abs(k - (int) odotusarvo);
81  
82          parvo = binomi.binomialTest(n, ylaraja, p, AlternativeHypothesis.GREATER_THAN)
83                  + binomi.binomialTest(n, alaraja, p, AlternativeHypothesis.LESS_THAN);
84  
85          System.out.println("parvo " + parvo);
86  
87          return parvo;
88      }
