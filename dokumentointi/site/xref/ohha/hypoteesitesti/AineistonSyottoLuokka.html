80       * @param jakauma käyttäjän syöttämä jakauma, joka on muutettu integeriksi

82      public static void syotaDiskreettiAineisto(int jakauma) {
83          Scanner lukija = new Scanner(System.in);
84  
85          System.out.println("Aineiston tiedot.");
86          System.out.println("Syötä aineisto:");
87  
88          int n = 0;
89          int k = -1;
90          double p = -1;
91  
92          if (jakauma == 1) {
93              while (n <= 0) {
94  
95                  System.out.print("Otoskoko: n = ");
96                  n = Integer.parseInt(lukija.nextLine());
97  
98                  if (n <= 0) {
99                      System.out.println("Syötteesi ei kepaa. Valitse n > 0.");
100                 }
101             }
102         }
103 
104         while (k > n || k < 0) {
105 
106             System.out.print("Onnistumisten lukumäärä: k = ");
107             k = Integer.parseInt(lukija.nextLine());
108             if (jakauma != 1) {
109                 n = Math.max(k, n);
110             }
111             if (k > n || k < 0) {
112                 System.out.println("Syötteesi ei kelpaa. Valitse k >= 0.");
113             }
114 
115         }
116 
117         while (p < 0 || p > 1) {
118 
119             System.out.print("Nollahypoteesi H0: p = ");
120             p = Double.parseDouble(lukija.nextLine());
121 
122             if (p < 0 || p > 1) {
123                 System.out.println("Syötteesi ei kelpaa. Valitse 0 < p < 1.");
124             }
125         }
126 
127         int suunta = 0;
128 
129         while (suunta == 0) {
130             System.out.println("Vastahypoteesi. Syötä valintaasi vastaava numero. H1:");
131             System.out.println("1) p < " + p);
132             System.out.println("2) p > " + p);
133             System.out.println("3) Ei vastahypoteesia.");
134             suunta = Integer.parseInt(lukija.nextLine());
135 
136             if (suunta != 1 && suunta != 2 && suunta != 3) {
137                 System.out.println("Syötteesi ei kelpaa. Syötä joko 1, 2, tai 3.");
138                 suunta = 0;
139             }
140         }
141 
142         JakaumanTyyppi jakaumantyyppi = new JakaumanTyyppi(jakauma);
143         Testaaja testaaja = new Testaaja(jakaumantyyppi);
144 
145         testaaja.teeTesti(n, k, p, suunta);
