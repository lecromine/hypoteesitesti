13      public double ptnf(int n, int k, double p) {
14          if (p < 0 || p > 1 || n < k || n < 0 || k < 0) {
15              return -1;
16          } else if (k == 0 && n != 0) {
17              return Math.pow(1 - p, n);
18          } else if (n == 0) {
19              return -1;
20          }
21  
22          return kombinaatio(n, k) * Math.pow(p, k) * Math.pow(1 - p, n - k);
23      }
24  
25      public double kombinaatio(double n, double k) {
26          return permutaatio(n) / (permutaatio(k) * permutaatio(n - k));
27      }
28  
29      public double permutaatio(double i) {
30          if (i == 1 || i == 0) {
31              return 1;
32          }
33  
34          return i * permutaatio(i - 1);
35      }
