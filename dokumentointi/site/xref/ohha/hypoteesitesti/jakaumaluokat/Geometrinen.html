3   import org.apache.commons.math3.distribution.GeometricDistribution;
4   
5   /**
6    * Tämä luokka sisältää kaikki geometriseen jakaumaan liittyvät tiedot.
7    * Testauslogiikka säilytetään joko täällä tai siirretään Testaajaan: riippuu
8    * siitä, kuinka pitkä testistä tulee.
9    *
10   * @author Lecromine
11   */
12  public class Geometrinen {
13  
14      private int jakaumaNumero;
15      private double parvo = 0;
16  
17      public Geometrinen() {
18  
19          this.jakaumaNumero = 6;
20      }
21  
22      /**
23       *
24       * Testin suorittaminen geometrisella aineistolla käyttäjän antamilla
25       * arvoilla.
26       *
27       * @param k kokeiden lukumäärä + ensimmäinen onnistuminen
28       * @param p nollahypoteesia vastaava parametri
29       * @param suunta suunta: kaksisuuntainen, suurempi kuin, pienempi kuin
30       * @return parvo testin tuloksena saatu p-arvo, jonka avulla voidaan tehdä
31       * johtopäätöksiä
32       */
33      public double geometrinenTesti(int k, double p, int suunta) {
34  
35          if (k < 0 || p < 0 || p > 1 || suunta < 1 || suunta > 3) {
36              return -1;
37          }
38  
39          // kaksisuuntainen asetelma H0: p = p0
40          GeometricDistribution geom = new GeometricDistribution(p);
41  
42          double expected = (1 - p) / p;
43          int ylaraja = (int) expected - (int) Math.abs(k - expected); // kriittinen alue: [ylaraja, n]
44  
45          // esim. i = 10. H0: p>0.5 H1: p<0.5
46          // 
47          if (suunta == 2 || suunta == 3) {
48  
49              ylaraja = 0;
50  
51              for (int i = ylaraja; i <= k; i++) {
52                  parvo = parvo + geom.probability(i);
53              }
54          }
55  
56          return parvo;
57  
58      }
59  
60      public int getjakaumaNumero() {
61          return this.jakaumaNumero;
62      }
63  
64  }
