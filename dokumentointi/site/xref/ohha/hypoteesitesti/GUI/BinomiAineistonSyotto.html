155     private void valmisNappiActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_valmisNappiActionPerformed
156 
157         if (nKentta.getText().isEmpty()
158                 || kKentta.getText().isEmpty()
159                 || pKentta.getText().isEmpty()
160                 || Integer.parseInt(nKentta.getText()) < 1
161                 || Integer.parseInt(kKentta.getText()) < 0
162                 || Double.parseDouble(pKentta.getText()) < 0
163                 || Integer.parseInt(nKentta.getText()) < Integer.parseInt(kKentta.getText())
164                 || Double.parseDouble(pKentta.getText()) > 1
165                 || suuntaValikko.getSelectedItem().equals(" ")) {
166             new UlinaRuutu().setVisible(true);
167         } else {
168             
169             BinomiAineistonTulokset tulokset = new BinomiAineistonTulokset();
170             
171             tulokset.suunnanMaaritys(Integer.parseInt(nKentta.getText()), 
172                     Integer.parseInt(kKentta.getText()), 
173                     Double.parseDouble(pKentta.getText()), 
174                     suuntaValikko.getSelectedItem().toString());
175             
176             tulokset.setVisible(true);
177             
178         }
179 
180     }//GEN-LAST:event_valmisNappiActionPerformed
