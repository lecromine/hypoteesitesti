13      public void tulosta(double ka, int n, double p, double s, String suunta, String testi) {
14          
15          maaritaTestattavaParametri(testi);
16          
17          int testiNumero = 0;
18          int suuntaNumero = 0;
19          
20          switch (testi) {
21              case "Odotusarvo (varianssi tunnettu)":
22                  testiNumero = 1;
23                  break;
24              case "Odotusarvo (varianssi tuntematon)":
25                  testiNumero = 2;
26                  break;
27              case "Varianssi":
28                  testiNumero = 3;
29                  break;
30              default:
31                  break;
32          }
33          
34          switch (suunta) {
35              case "<":
36                  suuntaNumero = 1;
37                  suunta1.setText("<");
38                  
39                  break;
40              case ">":
41                  suuntaNumero = 2;
42                  suunta1.setText(">");
43                  break;
44              case "=":
45                  suuntaNumero = 3;
46                  suunta1.setText("=");
47                  break;
48              default:
49                  break;
50          }
51          
52  
53          DecimalFormat df = new DecimalFormat("##.#####");
54  
55          sValue.setText(s + "");
56          nValue.setText(n + "");
57          pValue.setText(p + "");
58          
59          if (ka != -1) {
60              kaValue.setText(ka + "");
61          }
62  
63          Testaaja testaaja = new Testaaja();
64          double parvo = testaaja.normaaliTesti(n, ka, s, p, suuntaNumero, testiNumero);
65          parvoValue.setText("" + parvo);
66  
67          if (parvo < 0.1) {
68              if (parvo < 0.05) {
69                  if (parvo < 0.01) {
70  
71                      tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
72                      tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
73                      tulosrivi3.setText("Nollahypoteesin todennäköisyys on vähemmän kuin 1%");
74                      tulosrivi4.setText("joten se hylätään tasolla 1%");
75  
76                  } else if (parvo > 0.01) {
77                      tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
78                      tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
79                      tulosrivi3.setText("Nollahypoteesi hylätään");
80                      tulosrivi4.setText("5% merkitsevyystasolla mutta hyväksytään 1% tasolla.");
81                  }
82              } else {
83  
84                  tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
85                  tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
86                  tulosrivi3.setText("Nollahypoteesi hylätään");
87                  tulosrivi4.setText("10% merkitsevyystasolla mutta hyväksytään 5% tasolla");
88  
89              }
90          } else if (parvo > 0.1) {
91  
92              tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
93              tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
94              tulosrivi3.setText("Nollahypoteesi hyväksytään");
95              tulosrivi4.setText("10% merkitsevyystasolla");
96  
97          }
98      }

100     public void maaritaTestattavaParametri(String testi) {
101         if (testi.equals("Odotusarvo (varianssi tunnettu)")) {
102             otoska.setText("Otoskeskiarvo");
103             nollahypo.setText("Nollahypoteesi H\u2080: \u00B5\u2080");
104             keskihaj.setText("Varianssi");
105         } else if (testi.equals("Odotusarvo (varianssi tuntematon)")) {
106             keskihaj.setText("Keskihajonta");
107             otoska.setText("Otoskeskiarvo");
108             nollahypo.setText("Nollahypoteesi H\u2080: \u00B5\u2080");
109         } else if (testi.equals("Varianssi")) {
110             keskihaj.setText("Keskihajonta");
111             otoska.setText("");
112             nollahypo.setText("Nollahypoteesi H\u2080: \u03C3\u2080");
113         }
114     }
