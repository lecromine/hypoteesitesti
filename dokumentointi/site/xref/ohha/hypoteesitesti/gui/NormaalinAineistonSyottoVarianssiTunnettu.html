184     private void valmisNappiActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_valmisNappiActionPerformed
185 
186         if (Double.parseDouble(kaKentta.getText()) <= 0
187                 || Double.parseDouble(varKentta.getText()) < 0
188                 || Double.parseDouble(pKentta.getText()) < 0
189                 || Integer.parseInt(nKentta.getText()) < 1) {
190             new UlinaRuutu().setVisible(true);
191         } else {
192 
193             NormaalinAineistonTulokset tulokset = new NormaalinAineistonTulokset();
194 
195             tulokset.tulosta(Double.parseDouble(kaKentta.getText()),
196                     Integer.parseInt(nKentta.getText()),
197                     Double.parseDouble(pKentta.getText()),
198                     Double.parseDouble(varKentta.getText()),
199                     "Odotusarvo (varianssi tunnettu)");
200 
201             tulokset.setVisible(true);
202         }
203 
204     }//GEN-LAST:event_valmisNappiActionPerformed
