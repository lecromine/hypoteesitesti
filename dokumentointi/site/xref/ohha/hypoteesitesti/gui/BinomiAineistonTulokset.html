13      public void suunnanMaaritys(int n, int k, double p, String suunta) {
14          DecimalFormat df = new DecimalFormat("##.###");
15          int suuntaNumero = 0;
16          
17          nValue.setText(n + "");
18          kValue.setText(k + "");
19          pValue0.setText(p + "");
20          
21          if (suunta.equals("<")) {
22              suuntaNumero = 1;
23              suunta1.setText("<");
24              suunta2.setText(">");
25              pValue1.setText(p + "");
26          } else if (suunta.equals(">")) {
27              suuntaNumero = 2;
28              suunta1.setText(">");
29              suunta2.setText("<");
30              pValue1.setText(p + "");
31          } else if (suunta.equals("=")) {
32              suuntaNumero = 3;
33              suunta1.setText("=");
34              suunta2.setText("");
35              pValue1.setText("");
36  
37          }
38          
39          Testaaja testaaja = new Testaaja();
40          double parvo = testaaja.teeBinomiTesti(n, k, p, suuntaNumero);
41          parvoValue.setText("" + df.format(parvo));
42  
43          if (parvo < 0.1) {
44              if (parvo < 0.05) {
45                  if (parvo < 0.01) {
46  
47                      tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
48                      tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
49                      tulosrivi3.setText("Nollahypoteesi hylätään");
50                      tulosrivi4.setText("1% merkitsevyystasolla ");
51  
52                  } else if (parvo > 0.01) {
53                      tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
54                      tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
55                      tulosrivi3.setText("Nollahypoteesi hyväksytään");
56                      tulosrivi4.setText("5% merkitsevyystasolla mutta hylätään 1% tasolla.");
57                  }
58              } else {
59  
60                  tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
61                  tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
62                  tulosrivi3.setText("Nollahypoteesi hylätään");
63                  tulosrivi4.setText("10% merkitsevyystasolla mutta hyväksytään 5% tasolla");
64  
65              }
66          } else if (parvo > 0.1) {
67  
68              tulosrivi1.setText("Todennäköisyys saada oheinen aineisto");
69              tulosrivi2.setText("nollahypoteesin pätiessä on " + df.format(parvo * 100) + "%.");
70              tulosrivi3.setText("Nollahypoteesi hyväksytään");
71              tulosrivi4.setText("10% merkitsevyystasolla");
72  
73          }
74      }
