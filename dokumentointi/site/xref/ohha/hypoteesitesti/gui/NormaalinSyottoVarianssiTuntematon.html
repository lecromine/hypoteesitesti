238     private void valmisNappiActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_valmisNappiActionPerformed
239         
240         try {
241             double s = Double.parseDouble(sKentta.getText());
242             double p = Double.parseDouble(pKentta.getText());
243             double ka = Double.parseDouble(kaKentta.getText());
244             int n = Integer.parseInt(nKentta.getText());
245         } catch (final NumberFormatException e) {
246             new UlinaRuutuNormaalille().setVisible(true);
247         }
248         
249         if (sKentta.getText().isEmpty() == true 
250                 || pKentta.getText().isEmpty() == true
251                 || nKentta.getText().isEmpty() == true
252                 || Double.parseDouble(kaKentta.getText()) <= 0
253                 || Double.parseDouble(sKentta.getText()) <= 0
254                 || Double.parseDouble(pKentta.getText()) <= 0
255                 || Integer.parseInt(nKentta.getText()) < 1
256                 || suuntaValikko.getSelectedItem().equals(" ")) {
257             new UlinaRuutuBinomille().setVisible(true);
258         } else {
259 
260             NormaalinTulokset tulokset = new NormaalinTulokset();
261 
262             tulokset.tulosta(Double.parseDouble(kaKentta.getText()),
263                     Integer.parseInt(nKentta.getText()),
264                     Double.parseDouble(pKentta.getText()),
265                     Double.parseDouble(sKentta.getText()),
266                     suuntaValikko.getSelectedItem().toString(),
267                     "Odotusarvo (varianssi tuntematon)");
268 
269             tulokset.setVisible(true);
270         }
271 
272     }//GEN-LAST:event_valmisNappiActionPerformed
